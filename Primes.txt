def find_primes():
    n = int(input("Enter the upper limit for the range you want to find primes till : "))
    primes=[]
    for i in range(2,n+1):
        is_prime = True
        for j in range(2,(i//2)+1):
            if i % j == 0:
                is_prime =  False
                break
        if is_prime:
            primes.append(i)
    return(primes)
primes = find_primes()
print(primes)
