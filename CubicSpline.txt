import numpy as np
from scipy.interpolate import CubicSpline
import matplotlib.pyplot as plt
x = np.linspace(1,6,10)
print(len(x))
y = np.array([1,3,2,4,2.4,5,6,5,3.4,4.5])
print(len(y))
CS = CubicSpline(x,y)
xc = np.linspace(min(x),max(y),100)
yc = CS(xc)
plt.figure(figsize=(7,6))
plt.plot(x,y,'o',label = "Data Points")
plt.plot(xc,yc,label = "CubicSpline Interpolation")
plt.xlabel('x')
plt.ylabel('y')
plt.show()
